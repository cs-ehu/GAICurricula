182 	/**
183 	 * Inicializa el botón para procesar el pom de un proyecto y trata el pom después de validar los campos de texto necesarios de la aplicación.
184 	 * 
185 	 * @param xqm La lógica utilizada para procesar el pom.
186 	 */
187 	public void btnProcesarPomInit(XQueryMethods xqm) {
188 		JButton btnProcesarPom = new JButton("Procesar POM");
189 		btnProcesarPom.addActionListener(new ActionListener() {
190 			public void actionPerformed(ActionEvent e) {
191 				if(pomField.getText().isEmpty() || htmlField.getText().isEmpty() || fileHTMLField.getText().isEmpty()) {
192 					JOptionPane.showMessageDialog(null,
193 							"El usuario debe rellenar todos los campos para poder \nproducir el HTML con la información del POM.",
194 							"Rellene todos los campos",
195 							JOptionPane.WARNING_MESSAGE);
196 				} else {
197 					if(!fileHTMLField.getText().endsWith(".html")) {
198 						JOptionPane.showMessageDialog(null,
199 								"El nombre de fichero HTML que se va a producir debe tener \nuna extensión adecuada.",
200 								"Rellene el campo del fichero a generar adecuadamente",
201 								JOptionPane.WARNING_MESSAGE);
202 					} else {
203 						try {
204 							xqm.generaXQPOM(pomField.getText().toString(), htmlField.getText().toString() + "\\" + fileHTMLField.getText().toString());
205 							int confirmed = JOptionPane.showConfirmDialog(null, 
206 									"¡El fichero HTML ha sido generado con éxito! \n¿Desea verlo? si presiona sí se le abrirá.", "¡Archivo HTML generado!",
207 									JOptionPane.YES_NO_OPTION);
208 	
209 							if (confirmed == JOptionPane.YES_OPTION) {
210 								Desktop desktop = Desktop.getDesktop();
211 								File file = new File(htmlField.getText().toString() + "//" + fileHTMLField.getText().toString());
212 								if(file.exists()) desktop.open(file);
213 							}
214 						} catch(XQException | IOException ex2) {
215 							JOptionPane.showMessageDialog(null,
216 									"No se ha generado el fichero HTML. \nEl usuario debe rellenar todos los campos adecuadamente.",
217 									"Error",
218 									JOptionPane.ERROR_MESSAGE);
219 						}
220 					}
221 				}
222 			}
223 		});
224 		btnProcesarPom.setBounds(72, 220, 137, 23);
225 		frmHtmlpom.getContentPane().add(btnProcesarPom);
226 	}
